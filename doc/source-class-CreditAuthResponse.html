276     /**
277      * Validates CreditAuth object before attempting transaction
278      * 
279      * @throws Exception on invalid data
280      */
281     protected function validate()
282     {
283         if($this->swipe)
284             return true;
285 
286         $cardTypes = [
287             3 => 'AMEX',
288             4 => 'VISA',
289             5 => 'MC',
290             6 => 'DISC'
291         ];
292 
293         if(!in_array(substr($this->CardNbr, 0, 1), ['3','4','5','6']))
294             throw new Exception('Invalid card brand identifier: '.substr($this->CardNbr, 0, 1));
295         
296         $ex = [];
297 
298         if($this->ExpYear < 2000)
299             $this->ExpYear += 2000;
300         
301         switch($cardTypes[substr($this->CardNbr, 0, 1)])
302         {
303             case 'AMEX':
304                 if(strlen($this->CardNbr) != 15)
305                     $ex[] = 'Invalid AMEX card number length';
306                 if(strlen($this->CVV2) != 4)
307                     $ex[] = 'Invalid AMEX CVV length';
308                 break;
309 
310             case 'VISA':
311             case 'MC':
312             case 'DISC':
313                 if(strlen($this->CardNbr) != 16)
314                     $ex[] = 'Invalid '.$cardTypes[substr($this->CardNbr, 0, 1)].' card number length';
315                 if(strlen($this->CVV2) != 3)
316                     $ex[] = 'Invalid '.$cardTypes[substr($this->CardNbr, 0, 1)].' CVV length';
317                 break;
318 
319         }
320 
321         if($this->ExpYear.str_pad($this->ExpMonth, 2, '0', STR_PAD_LEFT) < date('Ym'))
322             $ex[] = 'Expired Card';
323 
324         if(!in_array(strlen($this->CardHolderZip), [5,9]))
325             $ex[] = 'Invalid Billing ZIP Code';
326 
327         if(count($ex) > 0)
328             throw new Exception(implode("\n", $ex));
329 
330         return true;
331     }

334     /**
335      * Child class specific function to check for Transaction failures
336      *
337      * @param  object $response Raw Heartland Portico Gateway response
338      *
339      * @return object|boolean
340      */
341     protected function checkFailure($response)
342     {
343         if(isset($response->{'Ver1.0'}->Transaction->CreditAuth->RspCode) && $response->{'Ver1.0'}->Transaction->CreditAuth->RspCode != 0)
344         {   
345             $failure = new FailureResponse($response, $this->request);
346             $failure->Success  = false;
347             $failure->FailType = 'Transaction';
348             $failure->RspCode  = $response->{'Ver1.0'}->Transaction->CreditAuth->RspCode;
349             $failure->RspText  = $response->{'Ver1.0'}->Transaction->CreditAuth->RspText;
350 
351             return $failure;
352         }else{
353             return false;
354         }
355     }
