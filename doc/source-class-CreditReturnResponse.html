29 /**
30  * CreditReturn extends portico to return partial or full amounts to previously completed, successful transaction  
31  * regardless of current batch status (but recommended only for Closed Batches).
32  *
33  * @package porticoAPI
34  *
35 */
36 class CreditReturn extends portico{
37     /**
38      * Array of required properties.
39      *
40      * @var array
41      */
42     protected $required = [
43         'CreditReturn' => [
44             'GatewayTxnId','Amt'
45         ]
46     ];
47     /**
48      * GatewayTxnId identifying transaction to return
49      * 
50      * @var string  
51      */
52     var $GatewayTxnId;      
53     /**
54      * Amount to return to cardholder
55      * 
56      * @var string  
57      */
58     var $Amt;
59 /*********************************************************************************************/
60 ###############################################################################################
61     /**
62      * Builds CreditReturn request and submit it to portico::Transaction
63      *
64      * @return CreditReturnResponse|FailureResponse formatted response from Heartland SOAP interface
65      */
66     public function doTransaction()
67     {
68         $this->initRequest();
69 
70         $this->setTransaction([
71             'CreditReturn' => [
72                 'Block1' => [
73                     'GatewayTxnId'  => $this->GatewayTxnId,
74                     'Amt'           => $this->Amt
75                 ]
76             ]
77         ]);
78         
79         return $this->Transaction();
80     }
81 }
